        // ---------- FUNCIONES DE CÁLCULO ----------

        function calcularIndiceAutodeterminacion(d) {

            const infFam = d.infFam || 3;

            const infSoc = d.infSoc || 3;

            const infEdu = d.infEdu || 3;

            const claridad = d.infClaridad || 3;

            const pesoFam = (6 - infFam) * 2;

            const pesoSoc = (6 - infSoc) * 2;

            const pesoEdu = (6 - infEdu) * 2;

            const pesoClar = claridad * 4;

            let score = Math.round(((pesoFam + pesoSoc + pesoEdu + pesoClar) / 44) * 100);

            return Math.min(100, Math.max(0, score));

        }

 

        function obtenerNivel(score) {

            if (score >= 75) return { texto: 'ALTA AUTODETERMINACIÓN', clase: 'text-green-600', diagnostico: 'El estudiante muestra una clara percepción de su propósito y baja interferencia de influencias externas.', recomendacion: 'Potenciar la autonomía mediante proyectos concretos.' };

            if (score >= 50) return { texto: 'AUTODETERMINACIÓN MODERADA', clase: 'text-yellow-600', diagnostico: 'Existe claridad pero las influencias externas generan dudas.', recomendacion: 'Trabajar en diferenciación entre deseos propios y ajenos.' };

            return { texto: 'BAJA AUTODETERMINACIÓN', clase: 'text-red-600', diagnostico: 'Predominan influencias externas y baja claridad.', recomendacion: 'Priorizar autoconocimiento y análisis de creencias limitantes.' };

        }
